00061 /* Clock configurations for the AT91SAM3S4-EK */
00062 #define CKGR_MUL_SHIFT         16
00063 #define CKGR_PLLCOUNT_SHIFT     8
00064 #define CKGR_DIV_SHIFT          0
00065 
00066 /* Clock configuration for the AT91SAM3S */
00067 static const ClockConfiguration clockConfigurations[] = {
00068 
00069     /* PCK = 24 MHz, MCK = 24 MHz
00070      * PCK = 12000000 * (7+1) / 2 / 2 = 24 MHz
00071      */
00072     {24, 24, (CKGR_PLLAR_STUCKTO1 | (7 << CKGR_MUL_SHIFT) \
00073         | (0x3f << CKGR_PLLCOUNT_SHIFT) \
00074         | (2 << CKGR_DIV_SHIFT)),
00075         ( PMC_MCKR_PRES_CLK_2 | PMC_MCKR_CSS_PLLA_CLK)},
00076     /* PCK = 48 MHz, MCK = 48 MHz
00077      * PCK = 12000000 * (7+1) / 1 / 2 = 48 MHz
00078      */
00079     {48, 48, (CKGR_PLLAR_STUCKTO1 | (7 << CKGR_MUL_SHIFT) \
00080         | (0x3f << CKGR_PLLCOUNT_SHIFT) \
00081         | (1 << CKGR_DIV_SHIFT)),
00082         ( PMC_MCKR_PRES_CLK_2 | PMC_MCKR_CSS_PLLA_CLK)},
00083     /* PCK = 64 MHz, MCK = 64 MHz
00084      * PCK = 12000000 * (15+1) / 3 / 1 = 64 MHz
00085      */
00086     {64, 64, (CKGR_PLLAR_STUCKTO1 | (15 << CKGR_MUL_SHIFT) \
00087         | (0x3f << CKGR_PLLCOUNT_SHIFT) \
00088         | (3 << CKGR_DIV_SHIFT)),
00089         ( PMC_MCKR_PRES_CLK | PMC_MCKR_CSS_PLLA_CLK)}
00090 };

00102 /**
00103  * \brief Sets the specified clock configuration.
00104  *
00105  * \param configuration  Index of the configuration to set.
00106  */
00107 void CLOCK_SetConfig(uint8_t configuration)
00108 {
00109     TRACE_DEBUG("Setting clock configuration #%d ... ", configuration);
00110     currentConfig = configuration;
00111 
00112     /* Switch to main oscillator in two operations */
00113     PMC->PMC_MCKR = (PMC->PMC_MCKR & (uint32_t)~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK;
00114     while ((PMC->PMC_SR & PMC_SR_MCKRDY) == 0);
00115 
00116     /* Configure PLL */
00117     PMC->CKGR_PLLAR = clockConfigurations[configuration].pllr;
00118     while ((PMC->PMC_SR & PMC_SR_LOCKA) == 0);
00119 
00120     /* Configure master clock in two operations */
00121     PMC->PMC_MCKR = (clockConfigurations[configuration].mckr & (uint32_t)~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK;
00122     while ((PMC->PMC_SR & PMC_SR_MCKRDY) == 0);
00123     PMC->PMC_MCKR = clockConfigurations[configuration].mckr;
00124     while ((PMC->PMC_SR & PMC_SR_MCKRDY) == 0);
00125 
00126     /* DBGU reconfiguration */
00127     UART_Configure(115200, clockConfigurations[configuration].mck*1000000);
00128     TRACE_DEBUG("done.\n\r");
00129 }
