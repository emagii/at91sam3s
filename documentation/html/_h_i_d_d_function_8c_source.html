00310 /**
00311  * Parse descriptors: Interface, Interrupt IN/OUT.
00312  * \param desc Pointer to descriptor list.
00313  * \param arg  Argument, pointer to HIDDParseData instance.
00314  */
00315 static uint32_t HIDDFunction_Parse(USBGenericDescriptor * pDesc,
00316                                    HIDDParseData * pArg)
00317 {
00318     /* Find HID Interface */
00319     if (pArg->pIfDesc == 0) {
00320         if (pDesc->bDescriptorType == USBGenericDescriptor_INTERFACE) {
00321             USBInterfaceDescriptor *pIf = (USBInterfaceDescriptor*)pDesc;
00322             /* Right interface for HID:
00323                    HID Class + at least 1 endpoint */
00324             if (pIf->bInterfaceClass == HIDInterfaceDescriptor_CLASS
00325                 && pIf->bNumEndpoints >= 1) {
00326                 /* Obtain new interface setting */
00327                 if (pArg->pHidd->bInterface == 0xFF) {
00328                     pArg->pHidd->bInterface = pIf->bInterfaceNumber;
00329                     pArg->pIfDesc = pIf;
00330                 }
00331                 /* Find specific interface setting */
00332                 else if (pArg->pHidd->bInterface == pIf->bInterfaceNumber) {
00333                     pArg->pIfDesc = pIf;
00334                 }
00335             }
00336         }
00337     }
00338     /* Interface end */
00339     else {
00340         /* Start another interface ? */
00341         if (pDesc->bDescriptorType == USBGenericDescriptor_INTERFACE) {
00342             /* Terminate the parse */
00343             return USBRC_PARTIAL_DONE;
00344         }
00345         /* Parse HID descriptor */
00346         else if (pDesc->bDescriptorType == HIDGenericDescriptor_HID) {
00347             pArg->pHidd->pHidDescriptor = (HIDDescriptor*)pDesc;
00348         }
00349         /* Parse endpoints */
00350         else if (pDesc->bDescriptorType == USBGenericDescriptor_ENDPOINT) {
00351             USBEndpointDescriptor *pEp = (USBEndpointDescriptor*)pDesc;
00352             if (pEp->bEndpointAddress & 0x80)
00353                 pArg->pHidd->bPipeIN = pEp->bEndpointAddress & 0x7F;
00354             else
00355                 pArg->pHidd->bPipeOUT = pEp->bEndpointAddress;
00356         }
00357 
00358         /* Check if all data is OK */
00359         if (pArg->pHidd->bInterface  != 0xFF
00360             && pArg->pHidd->bPipeIN  != 0xFF
00361             && pArg->pHidd->bPipeOUT != 0xFF)
00362             return USBRC_FINISHED;
00363     }
00364     return 0;
00365 }

00550 /**
00551  * Handles HID-specific SETUP request sent by the host.
00552  * \param pHidd Pointer to HIDDFunction instance.
00553  * \param request Pointer to a USBGenericRequest instance
00554  */
00555 uint32_t HIDDFunction_RequestHandler(HIDDFunction *pHidd,
00556                                      const USBGenericRequest *request)
00557 {
00558     uint32_t reqCode = (request->bmRequestType << 8)
00559                      | (request->bRequest);
00560 
00561     switch (reqCode) {
00562     /* Get_Descriptor */
00563     case USBGenericRequest_GETDESCRIPTOR|(0x81<<8):
00564         return HIDDFunction_GetDescriptor(
00565                     pHidd,
00566                     USBGetDescriptorRequest_GetDescriptorType(request),
00567                     USBGenericRequest_GetLength(request));
00568     /* Clear_Feature (EP) */
00569     case USBGenericRequest_CLEARFEATURE|(0x02<<8):
00570         if (USBFeatureRequest_GetFeatureSelector(request)
00571             == USBFeatureRequest_ENDPOINTHALT) {
00572             uint8_t ep = USBGenericRequest_GetEndpointNumber(request);
00573             if (USBD_IsHalted(ep)) {
00574                 /* Unhalt EP */
00575                 USBD_Unhalt(ep);
00576                 /* Restart Polling OUT */
00577                 if (ep == pHidd->bPipeOUT) {
00578                     HIDDFunction_StartPollingOutputs(pHidd);
00579                 }
00580                 /* and send a zero-length packet */
00581                 USBD_Write(0, 0, 0, 0, 0);
00582             }
00583             break; /* Handled success */
00584         }
00585         return USBRC_PARAM_ERR;
00586     /* Set_Descriptor */
00587     case USBGenericRequest_SETDESCRIPTOR|(0x01<<8):
00588         /* Optional, not implemented */
00589         USBD_Stall(0);
00590         break;
00591     /* Get_Idle */
00592     case (0xa1<<8)|HIDGenericRequest_GETIDLE:
00593         HIDDFunction_GetIdle(pHidd,
00594                              HIDReportRequest_GetReportId(request));
00595         break;
00596     /* Set_Idle */
00597     case (0x21<<8)|HIDGenericRequest_SETIDLE:
00598         HIDDFunction_SetIdle(pHidd,
00599                              HIDReportRequest_GetReportId(request),
00600                              HIDIdleRequest_GetIdleRate(request));
00601         break;
00602     /* Get_Report */
00603     case (0xa1<<8)|HIDGenericRequest_GETREPORT:
00604         HIDDFunction_GetReport(pHidd,
00605                                HIDReportRequest_GetReportType(request),
00606                                HIDReportRequest_GetReportId(request),
00607                                USBGenericRequest_GetLength(request));
00608         break;
00609     /* Set_Report */
00610     case (0x21<<8)|HIDGenericRequest_SETREPORT:
00611         HIDDFunction_SetReport(pHidd,
00612                                HIDReportRequest_GetReportType(request),
00613                                HIDReportRequest_GetReportId(request),
00614                                USBGenericRequest_GetLength(request));
00615         break;
00616     /* Get_Protocol */
00617     case (0xa1<<8)|HIDGenericRequest_SETPROTOCOL:
00618         pHidd->bProtocol = request->wValue;
00619         USBD_Write(0, 0, 0, 0, 0);
00620         break;
00621     /* Set_Protocol */
00622     case (0x21<<8)|HIDGenericRequest_GETPROTOCOL:
00623         USBD_Write(0, &pHidd->bProtocol, 1, 0, 0);
00624         break;
00625 
00626     default:

00628     }
