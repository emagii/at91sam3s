00070 /// Writes the current page of data on the NandFlash.
00071 /// Returns 0 if successful; otherwise returns 1.
00072 /// \param tnf  Pointer to a TranslatedNandFlash instance.
00073 //------------------------------------------------------------------------------
00074 static unsigned char FlushCurrentPage( Media *media )
00075 {
00076     // Make sure there is a page to flush
00077     if ( currentWritePage == -1 )
00078     {
00079         return 0 ;
00080     }
00081 
00082     TRACE_DEBUG("FlushCurrentPage(B#%d:P#%d)\n\r",
00083               currentWriteBlock, currentWritePage);
00084 
00085     // Write page
00086     if (TranslatedNandFlash_WritePage(TRANSLATED(media->interface),
00087                                       currentWriteBlock,
00088                                       currentWritePage,
00089                                       pageWriteBuffer,
00090                                       0)) {
00091 
00092         TRACE_ERROR("FlushCurrentPage: Failed to write page.\n\r");
00093         return 1;
00094     }
00095 
00096     // No current write page & block
00097     currentWriteBlock = -1;
00098     currentWritePage = -1;
00099 
00100     return 0;
00101 }

00104 /// Writes data at an unaligned (page-wise) address and size. The address is
00105 /// provided as the block & page number plus an offset. The data to write MUST
00106 /// NOT span more than one page.
00107 /// Returns 0 if the data has been written; 1 otherwise.
00108 /// \param media  Pointer to a nandflash Media instance.
00109 /// \param block  Number of the block to write.
00110 /// \param page  Number of the page to write.
00111 /// \param offset  Write offset.
00112 /// \param buffer  Data buffer.
00113 /// \param size  Number of bytes to write.
00114 //------------------------------------------------------------------------------
00115 static unsigned char UnalignedWritePage(
00116     Media *media,
00117     unsigned short block,
00118     unsigned short page,
00119     unsigned short offset,
00120     unsigned char *buffer,
00121     unsigned int size)
00122 {
00123     unsigned char error;
00124     unsigned short pageDataSize = NandFlashModel_GetPageDataSize(MODEL(media->interface));
00125     unsigned char writePage = ((size + offset) == pageDataSize);
00126 
00127     TRACE_DEBUG( "UnalignedWritePage(B%d:P%d@%d, %d)\n\r", (int)block, (int)page, (int)offset, (int)size ) ;
00128     assert( (size + offset) <= pageDataSize ) ; /* "UnalignedWrite: Write size and offset exceed page data size\n\r" */
00129 
00130     // If size is 0, return immediately
00131     if ( size == 0 )
00132     {
00133         return 0 ;
00134     }
00135 
00136     // If this is not the current page, flush the previous one
00137     if ((currentWriteBlock != block) || (currentWritePage != page))
00138     {
00139         // Flush and make page the new current page
00140         FlushCurrentPage(media);
00141         TRACE_DEBUG("Current write page: B#%d:P#%d\n\r", block, page);
00142         currentWriteBlock = block;
00143         currentWritePage = page;
00144 
00145         // Read existing page data in a temporary buffer if the page is not
00146         // entirely written
00147         if (size != pageDataSize) {
00148 
00149             error = TranslatedNandFlash_ReadPage(TRANSLATED(media->interface),
00150                                                  block,
00151                                                  page,
00152                                                  pageWriteBuffer,
00153                                                  0);
00154             if (error) {
00155 
00156                 TRACE_ERROR(
00157                           "UnalignedWrite: Could not read existing page data\n\r");
00158                 return 1;
00159             }
00160         }
00161     }
00162 
00163     // Copy data in temporary buffer
00164     memcpy(&(pageWriteBuffer[offset]), buffer, size);
00165     // Update read buffer if necessary
00166     if ((currentReadPage == currentWritePage)
00167         && (currentReadBlock == currentWriteBlock)) {
00168 
00169         TRACE_DEBUG("Updating current read buffer\n\r");
00170         memcpy(&(pageReadBuffer[offset]), buffer, size);
00171     }
00172 
00173     // Flush page if it is complete
00174     if (writePage) {
00175 
00176         FlushCurrentPage(media);
00177     }
00178 
00179     return 0;
