00054 /// \param time  Wake-up time.

00056 void SUPC_EnableFlash( Supc* pSupc, uint32_t dwTime )
00057 {
00058     pSupc->SUPC_FWUTR = dwTime ;
00059     pSupc->SUPC_MR = SUPC_KEY | pSupc->SUPC_MR | AT91C_SUPC_FLASHON ;
