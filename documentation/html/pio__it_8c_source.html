00090 /**
00091  * \brief Handles all interrupts on the given PIO controller.
00092  * \param id  PIO controller ID.
00093  * \param pPio  PIO controller base address.
00094  */
00095 extern void PioInterruptHandler( uint32_t id, Pio *pPio )
00096 {
00097     uint32_t status;
00098     uint32_t i;
00099 
00100     /* Read PIO controller status */
00101     status = pPio->PIO_ISR;
00102     status &= pPio->PIO_IMR;
00103 
00104     /* Check pending events */
00105     if ( status != 0 )
00106     {
00107         TRACE_DEBUG( "PIO interrupt on PIO controller #%d\n\r", id ) ;
00108 
00109         /* Find triggering source */
00110         i = 0;
00111         while ( status != 0 )
00112         {
00113             /* There cannot be an unconfigured source enabled. */
00114             assert(i < _dwNumSources);
00115 
00116             /* Source is configured on the same controller */
00117             if (_aIntSources[i].pPin->id == id)
00118             {
00119                 /* Source has PIOs whose statuses have changed */
00120                 if ( (status & _aIntSources[i].pPin->mask) != 0 )
00121                 {
00122                     TRACE_DEBUG( "Interrupt source #%d triggered\n\r", i ) ;
00123 
00124                     _aIntSources[i].handler(_aIntSources[i].pPin);
00125                     status &= ~(_aIntSources[i].pPin->mask);
00126                 }
00127             }
00128             i++;
00129         }
00130     }
00131 }

00214 /**
00215  * Configures a PIO or a group of PIO to generate an interrupt on status
00216  * change. The provided interrupt handler will be called with the triggering
00217  * pin as its parameter (enabling different pin instances to share the same
00218  * handler).
00219  * \param pPin  Pointer to a Pin instance.
00220  * \param handler  Interrupt handler function pointer.
00221  */

00223 {
00224     Pio* pio ;
00225     InterruptSource* pSource ;
00226 
00227     TRACE_DEBUG( "PIO_ConfigureIt()\n\r" ) ;
00228 
00229     assert( pPin ) ;
00230     pio = pPin->pio ;
00231     assert( _dwNumSources < MAX_INTERRUPT_SOURCES ) ;
00232 
00233     /* Define new source */
00234     TRACE_DEBUG( "PIO_ConfigureIt: Defining new source #%d.\n\r",  _dwNumSources ) ;
00235 
00236     pSource = &(_aIntSources[_dwNumSources]) ;
00237     pSource->pPin = pPin ;
00238     pSource->handler = handler ;
00239     _dwNumSources++ ;
00240 
00241     /* PIO3 with additional interrupt support
00242      * Configure additional interrupt mode registers */
00243     if ( pPin->attribute & PIO_IT_AIME )
00244     {
00245         // enable additional interrupt mode
00246         pio->PIO_AIMER       = pPin->mask ;
00247 
00248         // if bit field of selected pin is 1, set as Rising Edge/High level detection event
00249         if ( pPin->attribute & PIO_IT_RE_OR_HL )
00250         {
00251             pio->PIO_REHLSR    = pPin->mask ;
00252         }
00253         else
00254         {
00255             pio->PIO_FELLSR     = pPin->mask;
00256         }
00257 
00258         /* if bit field of selected pin is 1, set as edge detection source */
00259         if (pPin->attribute & PIO_IT_EDGE)
00260             pio->PIO_ESR     = pPin->mask;
00261         else
00262             pio->PIO_LSR     = pPin->mask;
00263     }
00264     else
00265     {
00266         /* disable additional interrupt mode */
00267         pio->PIO_AIMDR       = pPin->mask;
00268     }
