00062 /**
00063  * \brief  Check if the device is virgin.
00064  *
00065  * \param managed  Pointer to a ManagedNandFlash instance.
00066  * \param spare    Pointer to allocated spare area (must be assigned)
00067  * \return 1 if a nandflash device is virgin (i.e. has never been used as a
00068  * managed nandflash); otherwise return 0.
00069  */
00070 static uint8_t IsDeviceVirgin( const struct ManagedNandFlash *managed, uint8_t* spare )
00071 {
00072     struct NandBlockStatus blockStatus;
00073     const struct NandSpareScheme *scheme =
00074                             NandFlashModel_GetScheme(MODEL(managed));
00075     uint16_t baseBlock = managed->baseBlock;
00076     uint8_t badBlockMarker;
00077 
00078     uint8_t error;
00079 
00080     assert( spare ) ; /* "ManagedNandFlash_IsDeviceVirgin: spare\n\r" */
00081 
00082     /* Read spare area of page #0. */
00083     error = RawNandFlash_ReadPage(RAW(managed), baseBlock, 0, 0, spare);
00084     assert( !error ) ; /* "ManagedNandFlash_IsDeviceVirgin: Failed to read page #0\n\r" */
00085 
00086     /* Retrieve bad block marker and block status from spare area*/
00087     NandSpareScheme_ReadBadBlockMarker(scheme, spare, &badBlockMarker);
00088     NandSpareScheme_ReadExtra(scheme, spare, &blockStatus, 4, 0);
00089 
00090     /* Check if block is marked as bad*/
00091     if ( badBlockMarker != 0xFF )
00092     {
00093         /* Device is not virgin, since page #0 is guaranteed to be good*/
00094         return 0 ;
00095     }
00096     /* If device is not virgin, then block status will be set to either
00097        FREE, DIRTY or LIVE */
00098     else
00099     {
00100         if ( blockStatus.status != NandBlockStatus_DEFAULT )
00101         {
00102             /* Device is not virgin */
00103             return 0 ;
00104         }
00105     }
00106 
00107     return 1 ;
00108 }

00110 /**
00111  * \brief  Check if the given block is bad.
00112  *
00113  * \param managed  Pointer to a ManagedNandFlash instance.
00114  * \param block Raw block to check.
00115  * \param spare  Pointer to allocated spare area (must be assigned)
00116  * \return 1 if a nandflash device is virgin (i.e. has never been used as a
00117  * managed nandflash); otherwise return 0.
00118  */
00119 static uint8_t CheckBlock( const struct ManagedNandFlash *managed, uint16_t block, uint8_t* spare )
00120 {
00121     uint8_t error ;
00122     uint32_t i ;
00123     uint8_t pageSpareSize = NandFlashModel_GetPageSpareSize( MODEL( managed ) ) ;
00124 
00125     assert( spare != NULL ) ; /* "ManagedNandFlash_CheckBlock: spare\n\r" */
00126 
00127     /* Read spare area of first page of block */
00128     error = RawNandFlash_ReadPage( RAW( managed ), block, 0, 0, spare ) ;
00129     if ( error )
00130     {
00131         TRACE_ERROR( "CheckBlock: Cannot read page #0 of block #%d\n\r", block ) ;
00132         return error ;
00133     }
00134 
00135     /* Make sure it is all 0xFF */
00136     for ( i=0 ; i < pageSpareSize ; i++ )
00137     {
00138         if ( spare[i] != 0xFF )
00139         {
00140             return BADBLOCK ;
00141         }
00142     }
00143 
00144     /* Read spare area of second page of block */
00145     error = RawNandFlash_ReadPage( RAW( managed ), block, 1, 0, spare ) ;
00146     if ( error )
00147     {
00148         TRACE_ERROR( "CheckBlock: Cannot read page #1 of block #%d\n\r", block ) ;
00149         return error ;
00150     }
00151 
00152     /* Make sure it is all 0xFF */
00153     for ( i=0 ; i < pageSpareSize ; i++ )
00154     {
00155         if ( spare[i] != 0xFF )
00156         {
00157             return BADBLOCK ;
00158         }
00159     }
00160 
00161     return GOODBLOCK ;
00162 }
