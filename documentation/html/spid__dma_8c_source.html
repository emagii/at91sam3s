00219 /// Initializes the Spid structure and the corresponding SPI & DMA hardware.
00220 /// The driver will uses DMA channel 0 for RX and DMA channel 1 for TX.
00221 /// The DMA channels are freed automatically when no SPI command processing.
00222 /// \param pSpid  Pointer to a Spid instance.
00223 /// \param pSpiHw  Associated SPI peripheral.
00224 /// \param spiId  SPI peripheral identifier.
00225 /// \return Always 0.
00226 //------------------------------------------------------------------------------
00227 unsigned char SPID_Configure(Spid *pSpid, AT91S_SPI *pSpiHw, unsigned char spiId)
00228 {
00229     // Initialize the SPI structure
00230     pSpid->pSpiHw = pSpiHw;
00231     pSpid->spiId  = spiId;
00232     pSpid->semaphore = 1;
00233     pSpid->pCurrentCommand = 0;
00234 
00235     // Enable the SPI Peripheral
00236     PERIPH_ENABLE(pSpid->spiId);
00237     
00238     // Execute a software reset of the SPI twice
00239     WRITE_SPI(pSpiHw, SPI_CR, AT91C_SPI_SWRST);
00240     WRITE_SPI(pSpiHw, SPI_CR, AT91C_SPI_SWRST);
00241 
00242     // Configure SPI in Master Mode with No CS selected !!!
00243     WRITE_SPI(pSpiHw, SPI_MR, AT91C_SPI_MSTR | AT91C_SPI_MODFDIS | AT91C_SPI_PCS);
00244      
00245     // Disable the PDC transfer    
00246     WRITE_SPI(pSpiHw, SPI_PTCR, AT91C_PDC_RXTDIS | AT91C_PDC_TXTDIS);
00247 
00248     // Disable the SPI TX & RX
00249     WRITE_SPI(pSpiHw, SPI_CR, AT91C_SPI_SPIDIS);
00250 
00251     // Disable the SPI Peripheral
00252     PERIPH_DISABLE(pSpid->spiId);
00253 
00254     return 0;
00255 }

00277 //------------------------------------------------------------------------------
00278 /// Starts a SPI master transfer. This is a non blocking function. It will
00279 /// return as soon as the transfer is started.
00280 /// Returns 0 if the transfer has been started successfully; otherwise returns
00281 /// SPID_ERROR_LOCK is the driver is in use, or SPID_ERROR if the command is not
00282 /// valid.
00283 /// \param pSpid  Pointer to a Spid instance.
00284 /// \param pCommand Pointer to the SPI command to execute.
00285 //------------------------------------------------------------------------------
00286 unsigned char SPID_SendCommand(Spid *pSpid, SpidCmd *pCommand)
00287 {
00288     AT91S_SPI *pSpiHw = pSpid->pSpiHw;
00289      unsigned int spiMr;
00290          
00291      // Try to get the dataflash semaphore
00292      if (pSpid->semaphore == 0) {
00293     
00294          return SPID_ERROR_LOCK;
00295     }
00296      pSpid->semaphore--;
00297 
00298     // Enable the SPI Peripheral
00299     PERIPH_ENABLE(pSpid->spiId);
00300     
00301     // Disable PDC transmitter and receiver
00302     WRITE_SPI(pSpiHw, SPI_PTCR, AT91C_PDC_RXTDIS | AT91C_PDC_TXTDIS);
00303 
00304     // Write to the MR register
00305     spiMr = READ_SPI(pSpiHw, SPI_MR);
00306     spiMr |= AT91C_SPI_PCS;
00307     spiMr &= ~((1 << pCommand->spiCs) << 16);
00308     WRITE_SPI(pSpiHw, SPI_MR, spiMr);
00309 
00310     // Initialize DMA controller using channel 0 for RX, 1 for TX.
00311     configureDmaChannels();
00312     configureLinkList(pSpiHw, pCommand);
00313 
00314     // Initialize the callback
00315     pSpid->pCurrentCommand = pCommand;
00316     
00317     // Enable the SPI TX & RX
00318     WRITE_SPI(pSpiHw, SPI_CR, AT91C_SPI_SPIEN);
00319 
00320     // Start DMA 0(RX) && 1(TX)
00321     DMA_EnableChannels((1 << DMA_CHANNEL_0) | (1 << DMA_CHANNEL_1));
00322 
00323     // Enable DMA Interrupts
00324     DMA_EnableIt(  (DMA_CBTC << DMA_CHANNEL_0)
00325                  | (DMA_CBTC << DMA_CHANNEL_1));
00326 
00327     return 0;    
00328 }

00331 /// SPI DMA transfer ISR, Handle RX complete
00332 //------------------------------------------------------------------------------
00333 void SPID_Handler(Spid *pSpid)
00334 {
00335     unsigned int dmaStatus;
00336     SpidCmd *pSpidCmd = pSpid->pCurrentCommand;
00337     AT91S_SPI *pSpiHw = pSpid->pSpiHw;
00338 
00339     dmaStatus = DMA_GetStatus();
00340 
00341     if ((dmaStatus & AT91C_CBTC) == 0)
00342         return;
00343 
00344     if ((dmaStatus & (DMA_CBTC << DMA_CHANNEL_0)) == 0)
00345         return;
00346 
00347     // Disable the SPI TX & RX
00348     WRITE_SPI(pSpiHw, SPI_CR, AT91C_SPI_SPIDIS);
00349     // Disable the SPI Peripheral
00350     PERIPH_DISABLE(pSpid->spiId);
00351 
00352     // Disable DMA
00353     DMA_Disable();
00354     // Disable DMA Peripheral
00355     PERIPH_DISABLE(AT91C_ID_HDMA);
00356     
00357     // Release the dataflash semaphore
00358     pSpid->semaphore++;
00359         
00360     // Invoke the callback associated with the current command
00361     if (pSpidCmd && pSpidCmd->callback) {
00362     
00363         pSpidCmd->callback(0, pSpidCmd->pArgument);
00364     }
00365         
00366 }
