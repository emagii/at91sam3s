00030 /** \addtogroup flashd_module Flash Memory Interface
00031  * The flash driver manages the programming, erasing, locking and unlocking sequences
00032  * with dedicated commands.
00033  *
00034  * To implement flash programing operation, the user has to follow these few steps :
00035  * <ul>
00036  * <li>Configue flash wait states to initializes the flash. </li>
00037  * <li>Checks whether a region to be programmed is locked. </li>
00038  * <li>Unlocks the user region to be programmed if the region have locked before.</li>
00039  * <li>Erases the user page before program (optional).</li>
00040  * <li>Writes the user page from the page buffer.</li>
00041  * <li>Locks the region of programmed area if any.</li>
00042  * </ul>
00043  *
00044  * Writing 8-bit and 16-bit data is not allowed and may lead to unpredictable data corruption.
00045  * A check of this validity and padding for 32-bit alignment should be done in write algorithm.
00046 
00047  * Lock/unlock range associated with the user address range is automatically translated.
00048  *
00049  * This security bit can be enabled through the command "Set General Purpose NVM Bit 0".
00050  *
00051  * A 128-bit factory programmed unique ID could be read to serve several purposes.
00052  *
00053  * The driver accesses the flash memory by calling the lowlevel module provided in \ref efc_module.
00054  * For more accurate information, please look at the EEFC section of the Datasheet.
00055  *
00056  * Related files :\n
00057  * \ref flashd.c\n
00058  * \ref flashd.h.\n
00059  * \ref efc.c\n
00060  * \ref efc.h.\n
00061 */

00102  * \brief Computes the lock range associated with the given address range.
00103  *
00104  * \param dwStart  Start address of lock range.
00105  * \param dwEnd  End address of lock range.
00106  * \param pdwActualStart  Actual start address of lock range.
00107  * \param pdwActualEnd  Actual end address of lock range.
00108  */

00110 {
00111     Efc* pStartEfc ;
00112     Efc* pEndEfc ;
00113     uint16_t wStartPage ;
00114     uint16_t wEndPage ;
00115     uint16_t wNumPagesInRegion ;
00116     uint16_t wActualStartPage ;
00117     uint16_t wActualEndPage ;
00118 
00119     // Convert start and end address in page numbers
00120     EFC_TranslateAddress( &pStartEfc, dwStart, &wStartPage, 0 ) ;
00121     EFC_TranslateAddress( &pEndEfc, dwEnd, &wEndPage, 0 ) ;
00122 
00123     // Find out the first page of the first region to lock
00124     wNumPagesInRegion = IFLASH_LOCK_REGION_SIZE / IFLASH_PAGE_SIZE ;
00125     wActualStartPage = wStartPage - (wStartPage % wNumPagesInRegion) ;
00126     wActualEndPage = wEndPage ;
00127 
00128     if ( (wEndPage % wNumPagesInRegion) != 0 )
00129     {
00130         wActualEndPage += wNumPagesInRegion - (wEndPage % wNumPagesInRegion) ;
00131     }
00132     // Store actual page numbers
00133     EFC_ComputeAddress( pStartEfc, wActualStartPage, 0, pdwActualStart ) ;
00134     EFC_ComputeAddress( pEndEfc, wActualEndPage, 0, pdwActualEnd ) ;
00135     TRACE_DEBUG( "Actual lock range is 0x%06X - 0x%06X\n\r", *pdwActualStart, *pdwActualEnd ) ;
00136 }

00194 /**
00195  * \brief Writes a data buffer in the internal flash
00196  *
00197  * \note This function works in polling mode, and thus only returns when the
00198  * data has been effectively written.
00199  * \param address  Write address.
00200  * \param pBuffer  Data buffer.
00201  * \param size  Size of data buffer in bytes.
00202  * \return 0 if successful, otherwise returns an error code.
00203  */

00205 {
00206     Efc* pEfc ;
00207     uint16_t page ;
00208     uint16_t offset ;
00209     uint32_t writeSize ;
00210     uint32_t pageAddress ;
00211     uint16_t padding ;
00212     uint32_t dwError ;
00213     uint32_t sizeTmp ;
00214     uint32_t *pAlignedDestination ;
00215     uint32_t *pAlignedSource ;
00216 
00217     assert( pvBuffer ) ;
00218     assert( dwAddress >=IFLASH_ADDR ) ;
00219     assert( (dwAddress + dwSize) <= (IFLASH_ADDR + IFLASH_SIZE) ) ;
00220 
00221     /* Translate write address */
00222     EFC_TranslateAddress( &pEfc, dwAddress, &page, &offset ) ;
00223 
00224     /* Write all pages */
00225     while ( dwSize > 0 )
00226     {
00227         /* Copy data in temporary buffer to avoid alignment problems */
00228         writeSize = min((uint32_t)IFLASH_PAGE_SIZE - offset, dwSize ) ;
00229         EFC_ComputeAddress(pEfc, page, 0, &pageAddress ) ;
00230         padding = IFLASH_PAGE_SIZE - offset - writeSize ;
00231 
00232         /* Pre-buffer data */
00233         memcpy( _aucPageBuffer, (void *) pageAddress, offset);
00234 
00235         /* Buffer data */
00236         memcpy( _aucPageBuffer + offset, pvBuffer, writeSize);
00237 
00238         /* Post-buffer data */
00239         memcpy( _aucPageBuffer + offset + writeSize, (void *) (pageAddress + offset + writeSize), padding);
00240 
00241         /* Write page
00242          * Writing 8-bit and 16-bit data is not allowed and may lead to unpredictable data corruption
00243          */
00244         pAlignedDestination = (uint32_t*)pageAddress ;
00245         pAlignedSource = (uint32_t*)_adwPageBuffer ;
00246         sizeTmp = IFLASH_PAGE_SIZE ;
00247 
00248         while ( sizeTmp >= 4 )
00249         {
00250             *pAlignedDestination++ = *pAlignedSource++;
00251             sizeTmp -= 4;
00252         }
00253 
00254         /* Send writing command */
00255         dwError = EFC_PerformCommand( pEfc, EFC_FCMD_EWP, page, _dwUseIAP ) ;
00256         if ( dwError )
00257         {
00258             return dwError ;
00259         }
00260 
00261         /* Progression */
00262         dwAddress += IFLASH_PAGE_SIZE ;
00263         pvBuffer = (void *)((uint32_t) pvBuffer + writeSize) ;
00264         dwSize -= writeSize ;
00265         page++;
00266         offset = 0;
00267     }
00268 
00269     return 0 ;

00368  */

00370 {
00371     Efc *pEfc ;
00372     uint16_t startPage, endPage ;
00373     uint8_t startRegion, endRegion ;
00374     uint32_t numPagesInRegion ;
00375     uint32_t status ;
00376     uint32_t dwError ;
00377     uint32_t numLockedRegions = 0 ;
00378 
00379     assert( end >= start ) ;
00380     assert( (start >=IFLASH_ADDR) && (end <= IFLASH_ADDR + IFLASH_SIZE) ) ;
00381 
00382     // Compute page numbers
00383     EFC_TranslateAddress( &pEfc, start, &startPage, 0 ) ;
00384     EFC_TranslateAddress( 0, end, &endPage, 0 ) ;
00385 
00386     // Compute region numbers
00387     numPagesInRegion = IFLASH_LOCK_REGION_SIZE / IFLASH_PAGE_SIZE ;
00388     startRegion = startPage / numPagesInRegion ;
00389     endRegion = endPage / numPagesInRegion ;
00390     if ((endPage % numPagesInRegion) != 0)
00391     {
00392         endRegion++ ;
00393     }
00394 
00395     // Retrieve lock status
00396     dwError = EFC_PerformCommand( pEfc, EFC_FCMD_GLB, 0, _dwUseIAP ) ;
00397     assert( !dwError ) ;
00398     status = EFC_GetResult( pEfc ) ;
00399 
00400     // Check status of each involved region
00401     while ( startRegion < endRegion )
00402     {
00403         if ( (status & (1 << startRegion)) != 0 )
00404         {
00405             numLockedRegions++ ;
00406         }
00407         startRegion++ ;
00408     }
00409 
00410     return numLockedRegions ;
