00281  * Reports a change in which keys are currently pressed or release to the

00283  *
00284  * \param pressedKeys Pointer to an array of key codes indicating keys that have

00286  *                    HIDDKeyboardDriver_ChangeKeys().

00288  * \param releasedKeys Pointer to an array of key codes indicates keys that have
00289  *                     been released since the last call to
00290  *                     HIDDKeyboardDriver_ChangeKeys().

00292  * \return USBD_STATUS_SUCCESS if the report has been sent to the host;
00293  *        otherwise an error code.
00294  */
00295 uint32_t HIDDKeyboard_ChangeKeys(uint8_t *pressedKeys,
00296                                  uint8_t pressedKeysSize,
00297                                  uint8_t *releasedKeys,
00298                                  uint8_t releasedKeysSize)
00299 {
00300     HIDDKeyboard *pKbd = &hiddKeyboard;
00301     HIDDFunction *pHidd = &pKbd->hidDrv;
00302     HIDDKeyboardInputReport *pReport =
00303         (HIDDKeyboardInputReport *)pKbd->inputReports[0]->bData;
00304 
00305     /* Press keys */
00306     while (pressedKeysSize > 0) {
00307 
00308         /* Check if this is a standard or modifier key */
00309         if (HIDKeypad_IsModifierKey(*pressedKeys)) {
00310 
00311             /* Set the corresponding bit in the input report */
00312             HIDDKeyboardInputReport_PressModifierKey(
00313                 pReport,
00314                 *pressedKeys);
00315         }
00316         else {
00317 
00318             HIDDKeyboardInputReport_PressStandardKey(
00319                 pReport,
00320                 *pressedKeys);
00321         }
00322 
00323         pressedKeysSize--;
00324         pressedKeys++;
00325     }
00326 
00327     /* Release keys */
00328     while (releasedKeysSize > 0) {
00329 
00330         /* Check if this is a standard or modifier key */
00331         if (HIDKeypad_IsModifierKey(*releasedKeys)) {
00332 
00333             /* Set the corresponding bit in the input report */
00334             HIDDKeyboardInputReport_ReleaseModifierKey(
00335                 pReport,
00336                 *releasedKeys);
00337         }
00338         else {
00339 
00340             HIDDKeyboardInputReport_ReleaseStandardKey(
00341                 pReport,
00342                 *releasedKeys);
00343         }
00344 
00345         releasedKeysSize--;
00346         releasedKeys++;
00347     }
00348 
00349     /* Send input report through the interrupt IN endpoint */
00350     return USBD_Write(pHidd->bPipeIN,
