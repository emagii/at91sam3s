00030 /** \addtogroup acc_module Working with ACC
00031  * The ACC driver provides the interface to configure and use the ACC peripheral.\n
00032  *
00033  * It applies comparison on two inputs and gives a compare output.
00034  *
00035  * To Enable a ACC Comparison,the user has to follow these few steps:
00036  * <ul>
00037  * <li> Enable ACC peripheral clock by setting the corresponding bit in PMC_PCER1
00038  *      (PMC Peripheral Clock Enable Register 1)
00039  * </li>
00040  * <li> Reset the controller by asserting ACC_CR_SWRST in ACC_CR(ACC Control Register)
00041  </li>
00042  * <li> Configure the mode as following steps:  </li>
00043  * -#   Select inputs for SELMINUS and SELPLUS in ACC_MR (ACC Mode Register).
00044  * -#   Enable Analog Comparator by setting ACEN in ACC_MR.
00045  * -#   Configure Edge Type to detect different compare output.
00046  * </li>
00047  * <li> Wait until the automatic mask period expires by polling MASK bit in
00048  *      ACC_ISR.
00049  * </ul>
00050  *
00051  * For more accurate information, please look at the ACC section of the
00052  * Datasheet.
00053  *
00054  * Related files :\n
00055  * \ref acc.c\n
00056  * \ref acc.h\n
00057  */

00076 /**
00077  * \brief Initialize the ACC controller
00078  *
00079  * \param pAcc Pointer to an Acc instance.
00080  * \param idAcc ACC identifier
00081  * \param ucSelplus input connected to inp, 0~7
00082  * \param ucSelminus input connected to inm,0~7
00083  * \param wAc_en Analog comprator enabled/disabled
00084  * \param wEdge CF flag triggering mode
00085  * \param wInvert INVert comparator output,use pattern defined in the device header file
00086  */
00087 extern void ACC_Configure( Acc *pAcc, uint8_t idAcc, uint8_t ucSelplus, uint8_t ucSelminus,
00088                            uint16_t wAc_en, uint16_t wEdge, uint16_t wInvert )
00089 {
00090     /* Enable peripheral clock*/
00091     PMC->PMC_PCER1 = 1 << (idAcc - 32) ;
00092 
00093     /*  Reset the controller */
00094     pAcc->ACC_CR |= ACC_CR_SWRST ;
00095 
00096     /*  Write to the MR register */
00097     ACC_CfgModeReg( pAcc,
00098                     ( (ucSelplus<<ACC_MR_SELPLUS_Pos) & ACC_MR_SELPLUS_Msk ) |
00099                     ( (ucSelminus<<ACC_MR_SELMINUS_Pos) & ACC_MR_SELMINUS_Msk ) |
00100                     ( (wAc_en<<8) & ACC_MR_ACEN ) |
00101                     ( (wEdge<<ACC_MR_EDGETYP_Pos) & ACC_MR_EDGETYP_Msk ) |
00102                     ( (wInvert<<12) & ACC_MR_INV ) ) ;
00103     /* set hysteresis and current option*/
00104     pAcc->ACC_ACR = (ACC_ACR_ISEL_HISP | ((0x01 << ACC_ACR_HYST_Pos) & ACC_ACR_HYST_Msk));
00105 
00106     /* Automatic Output Masking Period*/
00107     while ( pAcc->ACC_ISR & (uint32_t)ACC_ISR_MASK ) ;
00108 }
