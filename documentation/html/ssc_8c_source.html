00078 /**
00079  * \brief Configures a SSC peripheral.If the divided clock is not used, the master
00080  * clock frequency can be set to 0.
00081  * \note The emitter and transmitter are disabled by this function.
00082  * \param bitRate  bit rate.
00083  * \param masterClock  master clock.
00084  */
00085 void SSC_Configure(uint32_t bitRate, uint32_t masterClock)
00086 {
00087     /* Enable SSC peripheral clock */
00088     PMC->PMC_PCER0 = 1 << ID_SSC;
00089 
00090     /* Reset, disable receiver & transmitter */
00091     SSC->SSC_CR = SSC_CR_RXDIS | SSC_CR_TXDIS | SSC_CR_SWRST;
00092 
00093     SSC->SSC_PTCR = SSC_PTCR_RXTDIS | SSC_PTCR_TXTDIS;
00094     /* Configure clock frequency */
00095     if (bitRate != 0) {
00096 
00097         SSC->SSC_CMR = masterClock / (2 * bitRate);
00098     }
00099     else {
00100 
00101         SSC->SSC_CMR = 0;
00102     }
00103 }
