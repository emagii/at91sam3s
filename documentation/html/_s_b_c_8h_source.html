00325 #define WORDB(bytes)            ((unsigned short) ((bytes[0] << 8) | bytes[1]))

00327 /** \brief  Converts a byte array to a dword value using the big endian format */

00330 
00331 /** \brief  Stores a dword value in a byte array, in big endian format */
00332 #define STORE_DWORDB(dword, bytes) \
00333     bytes[0] = (unsigned char) (((dword) >> 24) & 0xFF); \
00334     bytes[1] = (unsigned char) (((dword) >> 16) & 0xFF); \
00335     bytes[2] = (unsigned char) (((dword) >> 8) & 0xFF); \
00336     bytes[3] = (unsigned char) ((dword) & 0xFF);
00337 
00338 /** \brief  Stores a word value in a byte array, in big endian format */
00339 #define STORE_WORDB(word, bytes) \
00340     bytes[0] = (unsigned char) (((word) >> 8) & 0xFF); \
00341     bytes[1] = (unsigned char) ((word) & 0xFF);
