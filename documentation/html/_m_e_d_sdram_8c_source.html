00050 /// Do copy for 8-byte aligned data
00051 //------------------------------------------------------------------------------
00052 static void AlignedCopy(unsigned char * *src,
00053                         unsigned char * *dst,
00054                         unsigned int len)
00055 {
00056     unsigned int *src32, *dst32;
00057     src32 = (unsigned int*)*src;
00058     dst32 = (unsigned int*)*dst;
00059     for (;len > 0; len -= 8) {
00060         *dst32 ++ = *src32 ++;
00061         *dst32 ++ = *src32 ++;
00062     }
00063     *src = (unsigned char*)src32;
00064     *dst = (unsigned char*)dst32;
00065 }

00067 //------------------------------------------------------------------------------

00072                           unsigned int  len)
00073 {
00074     for (;len > 0; len --) {
00075         *(*dst) ++ = *(*src) ++;
