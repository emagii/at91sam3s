00137 //------------------------------------------------------------------------------
00138 /// Writes data on the At45 at the specified address. Only one page of
00139 /// data is written that way; if the address is not at the beginning of the
00140 /// page, the data is written starting from this address and wraps around to
00141 /// the beginning of the page.
00142 /// \param pAt45  Pointer to a At45 driver instance.
00143 /// \param pBuffer  Buffer containing the data to write.
00144 /// \param size  Number of bytes to write.
00145 /// \param address  Destination address on the At45.
00146 //------------------------------------------------------------------------------
00147 void AT45D_Write(
00148     At45 *pAt45,
00149     unsigned char *pBuffer,
00150     unsigned int size,
00151     unsigned int address) 
00152 {
00153     unsigned char error;
00154 
00155     SANITY_CHECK(pAt45);
00156     SANITY_CHECK(pBuffer);
00157     SANITY_CHECK(size <= pAt45->pDesc->pageSize);
00158 
00159     // Issue a page write through buffer 1 command
00160     error = AT45_SendCommand(pAt45, AT45_PAGE_WRITE_BUF1, 4, pBuffer, size, address, 0, 0);
00161     ASSERT(!error, "-F- AT45_Write: Could not issue command.\n\r");
00162 
00163     // Wait until the command is sent
00164     while (AT45_IsBusy(pAt45)) {
00165     
00166         AT45D_Wait(pAt45);
00167     }
00168 
00169     // Wait until the At45 becomes ready again
00170     AT45D_WaitReady(pAt45);
00171 }
