00042 //! \brief  Reads a specified amount of data from a flash memory
00043 //! \param  media    Pointer to a Media instance
00044 //! \param  address  Address of the data to read
00045 //! \param  data     Pointer to the buffer in which to store the retrieved
00046 //!                   data
00047 //! \param  length   Length of the buffer
00048 //! \param  callback Optional pointer to a callback function to invoke when
00049 //!                   the operation is finished
00050 //! \param  argument Optional pointer to an argument for the callback
00051 //! \return Operation result code
00052 //------------------------------------------------------------------------------
00053 static unsigned char FLA_Read(Media      *media,
00054                                uint32_t address,
00055                                void         *data,
00056                                uint32_t length,
00057                                MediaCallback   callback,
00058                                void         *argument)
00059 {
00060     unsigned char *source = (unsigned char *) address;
00061     unsigned char *dest = (unsigned char *) data;
00062 
00063     // Check that the media is ready
00064     if (media->state != MED_STATE_READY) {
00065 
00066         TRACE_INFO("Media busy\n\r");
00067         return MED_STATUS_BUSY;
00068     }
00069 
00070     // Check that the data to read is not too big
00071     if ((length + address) > media->size) {
00072 
00073         TRACE_WARNING("FLA_Read: Data too big\n\r");
00074         return MED_STATUS_ERROR;
00075     }
00076 
00077     // Enter Busy state
00078     media->state = MED_STATE_BUSY;
00079 
00080     // Read data
00081     while (length > 0) {
00082 
00083         *dest = *source;
00084 
00085         dest++;
00086         source++;
00087         length--;
00088     }
00089 
00090     // Leave the Busy state
00091     media->state = MED_STATE_READY;
00092 
00093     // Invoke callback
00094     if (callback != 0) {
00095 
00096         callback(argument, MED_STATUS_SUCCESS, 0, 0);
00097     }
00098 
00099     return MED_STATUS_SUCCESS;
00100 }

00140     if (length%4 != 0) {
00141

00145 
00146     // Check that the data to write is not too big
00147     if ((length + address) > media->size) {
00148 
00149         TRACE_WARNING("FLA_Write: Data too big\n\r");
00150         return MED_STATUS_ERROR;

00152 
00153     // Put the media in Busy state
00154     media->state = MED_STATE_BUSY;
00155 
00156     // Initialize the transfer descriptor
00157     media->transfer.data = data;
00158     media->transfer.address = address;
00159     media->transfer.length = length;
