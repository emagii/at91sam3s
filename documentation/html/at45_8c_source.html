00116 /// Sends a command to the dataflash through the SPI. The command is identified
00117 /// by its command code and the number of bytes to transfer (1 + number of
00118 /// address bytes + number of dummy bytes). If data needs to be received, then
00119 /// a data buffer must be provided.
00120 /// This function does not block; its optional callback will be invoked when
00121 /// the transfer completes.
00122 /// \param pAt45  Pointer to an At45 driver instance.
00123 /// \param cmd  Command code.
00124 /// \param cmdSize  Size of command code + address bytes + dummy bytes.
00125 /// \param pData  Data buffer.
00126 /// \param dataSize  Number of data bytes to send/receive.
00127 /// \param address  Address at which the command is performed if meaningful.
00128 /// \param callback  Optional callback to invoke at end of transfer.
00129 /// \param pArgument  Optional parameter to the callback function.
00130 //------------------------------------------------------------------------------
00131 unsigned char AT45_SendCommand(
00132     At45 *pAt45,
00133     unsigned char cmd,
00134     unsigned char cmdSize,
00135     unsigned char *pData,
00136     unsigned int dataSize,
00137     unsigned int address,
00138     SpidCallback callback,
00139     void *pArgument)
00140 {
00141     SpidCmd *pCommand;
00142     const At45Desc *pDesc = pAt45->pDesc;
00143     unsigned int dfAddress = 0;
00144 
00145     // Sanity checks
00146     ASSERT(pAt45, "AT45_Command: pAt45 is 0.\n\r");
00147     ASSERT(pDesc || (cmd == AT45_STATUS_READ),
00148            "AT45_Command: Device has no descriptor, only STATUS_READ command allowed\n\r");
00149 
00150     // Check if the SPI driver is available
00151     if (AT45_IsBusy(pAt45)) {
00152 
00153         return AT45_ERROR_LOCK;
00154     }
00155 
00156     // Compute command pattern
00157     pAt45->pCmdBuffer[0] = cmd;
00158 
00159     // Add address bytes if necessary
00160     if (cmdSize > 1) {
00161 
00162         ASSERT(pDesc, "AT45_Command: No descriptor for dataflash.\n\r");
00163         if (!configuredBinaryPage) {
00164             dfAddress =
00165                 ((address / (pDesc->pageSize)) << pDesc->pageOffset)
00166                  + (address % (pDesc->pageSize));
00167         }
00168         else {
00169             dfAddress = address;
00170         }
00171         // Write address bytes
00172         if (pDesc->pageNumber >= 16384) {
00173 
00174             pAt45->pCmdBuffer[1] = ((dfAddress & 0x0F000000) >> 24);
00175             pAt45->pCmdBuffer[2] = ((dfAddress & 0x00FF0000) >> 16);
00176             pAt45->pCmdBuffer[3] = ((dfAddress & 0x0000FF00) >> 8);
00177             pAt45->pCmdBuffer[4] = ((dfAddress & 0x000000FF) >> 0);
00178     
00179             if ((cmd != AT45_CONTINUOUS_READ) && (cmd != AT45_PAGE_READ)) {
00180     
00181                 cmdSize++;
00182             }
00183         }
00184         else {
00185 
00186             pAt45->pCmdBuffer[1] = ((dfAddress & 0x00FF0000) >> 16);
00187             pAt45->pCmdBuffer[2] = ((dfAddress & 0x0000FF00) >> 8);
00188             pAt45->pCmdBuffer[3] = ((dfAddress & 0x000000FF) >> 0);
00189         }
00190     }
00191 
00192     // Update the SPI Transfer descriptors
00193     pCommand = &(pAt45->command);
00194     pCommand->cmdSize = cmdSize;
00195     pCommand->pData = pData;
00196     pCommand->dataSize = dataSize;
00197     pCommand->callback = callback;
00198     pCommand->pArgument = pArgument;
00199 
00200     // Send Command and data through the SPI
00201     if (SPID_SendCommand(pAt45->pSpid, pCommand)) {
00202 
00203         return AT45_ERROR_SPI;
00204     }
00205 
00206     return 0;
00207 }

00210 /// This function returns the At45Desc structure corresponding to the device
00211 /// connected
00212 /// It automatically initializes pAt45->pDesc field structure.
00213 /// This function shall be called by the application before AT45_SendCommand.
00214 /// Returns 0 if successful; Otherwise, returns AT45_ERROR_LOCK if the At45
00215 /// driver is in use or AT45_ERROR_SPI if there was an error with the SPI driver.
00216 /// \param pAt45  Pointer to an AT45 driver instance.
00217 /// \param status  Device status register value.
00218 //------------------------------------------------------------------------------
00219 const At45Desc * AT45_FindDevice(At45 *pAt45, unsigned char status)
00220 {
00221     unsigned int i;
00222     unsigned char id = AT45_STATUS_ID(status);
00223 
00224     // Check if status is all one; in which case, it is assumed that no device
00225     // is connected
00226     if (status == 0xFF) {
00227 
00228         return 0;
00229     }
00230 
00231     // Look in device array
00232     i = 0;
00233     pAt45->pDesc = 0;
00234     while ((i < NUMDATAFLASH) && !(pAt45->pDesc)) {
00235 
00236         if (at45Devices[i].id == id) {
00237 
00238             pAt45->pDesc = &(at45Devices[i]);
00239         }
00240         i++;
00241     }
00242     configuredBinaryPage = AT45_STATUS_BINARY(status);
00243     return pAt45->pDesc;
00244 }
