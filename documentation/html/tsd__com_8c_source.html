00173  * \brief Interpolates the provided raw measurements using the previously calculated
00174  * slope. The resulting x and y coordinates are stored in an array.

00176  * \param pData  Raw measurement data, as returned by TSD_GetRawMeasurement().
00177  * \param pPoint  Array in which x and y will be stored.
00178  */

00180 {
00181     pPoint[0] = calibrationPoints[0].x
00182                 - (((int32_t) calibrationPoints[0].data[0] - (int32_t) pData[0]) * 1024)
00183                 / xSlope;
00184 
00185     pPoint[1] = calibrationPoints[0].y
00186                 - (((int32_t) calibrationPoints[0].data[1] - (int32_t) pData[1]) * 1024)
00187                 / ySlope;

00191         pPoint[0] = 0;

00197     }

00210 /**
00211  * \brief Performs the calibration process using the provided buffer to display
00212  * information.
00213  *
00214  * \return True if calibration was successful; otherwise false.
00215  */
00216 uint8_t TSDCom_Calibrate(void)
00217 {
00218     volatile uint32_t i; /* to keep the tempo with gcc code optimisation */
00219     int32_t slope1, slope2;
00220     CalibrationPoint measuredPoint;
00221     uint8_t xOk, yOk;
00222     int32_t xDiff, yDiff;
00223 
00224     /* Calibration setup */
00225     LCDD_Fill(COLOR_WHITE);
00226     LCDD_DrawString(30, 50, (uint8_t *)"LCD calibration", COLOR_BLACK);
00227     LCDD_DrawString(1, 140, (uint8_t *)"Touch the dots to\ncalibrate the screen", COLOR_DARKBLUE);
00228 
00229     /* Calibration points */
00230     for (i=0; i < 4; i++) {
00231 
00232         DrawCalibrationPoint(&calibrationPoints[i]);
00233 
00234         /* Wait for touch & end of conversion */
00235         TSD_WaitPenPressed();
00236         TSD_GetRawMeasurement(calibrationPoints[i].data);
00237         ClearCalibrationPoint(&calibrationPoints[i]);
00238 
00239         /* Wait for contact loss */
00240         TSD_WaitPenReleased();
00241     }
00242 
00243     /**
00244      * Calculate slopes using the calibration data
00245      * Theory behind those calculations:
00246      *   - We suppose the touchscreen measurements are linear, so the following equations are true (simple
00247      *     linear regression) for any two 'a' and 'b' points of the screen:
00248      *       dx = (a.data[0] - b.data[0]) / (a.x - b.x)
00249      *       dy = (a.data[1] - b.data[1]) / (a.y - b.y)
00250      *
00251      *   - We calculate dx and dy (called xslope and yslope here) using the calibration points.
00252      *
00253      *   - We can then use dx and dy to infer the position of a point 'p' given the measurements performed
00254      *     by the touchscreen ('c' is any of the calibration points):
00255      *       dx = (p.data[0] - c.data[0]) / (p.x - c.x)
00256      *       dy = (p.data[1] - c.data[1]) / (p.y - c.y)
00257      *     Thus:
00258      *       p.x = c.x - (p.data[0] - c.data[0]) / dx
00259      *       p.y = c.y - (p.data[1] - c.data[1]) / dy
00260      *
00261      *   - Since there are four calibration points, dx and dy can be calculated twice, so we average
00262      *     the two values.
00263      */
00264     slope1 = ((int32_t) calibrationPoints[0].data[0]) - ((int32_t) calibrationPoints[1].data[0]);
00265     slope1 *= 1024;
00266     slope1 /= ((int32_t) calibrationPoints[0].x) - ((int32_t) calibrationPoints[1].x);
00267     slope2 = ((int32_t) calibrationPoints[2].data[0]) - ((int32_t) calibrationPoints[3].data[0]);
00268     slope2 *= 1024;
00269     slope2 /= ((int32_t) calibrationPoints[2].x) - ((int32_t) calibrationPoints[3].x);
00270     xSlope = (slope1 + slope2) / 2;
00271 
00272     slope1 = ((int32_t) calibrationPoints[0].data[1]) - ((int32_t) calibrationPoints[2].data[1]);
00273     slope1 *= 1024;
00274     slope1 /= ((int32_t) calibrationPoints[0].y) - ((int32_t) calibrationPoints[2].y);
00275     slope2 = ((int32_t) calibrationPoints[1].data[1]) - ((int32_t) calibrationPoints[3].data[1]);
00276     slope2 *= 1024;
00277     slope2 /= ((int32_t) calibrationPoints[1].y) - ((int32_t) calibrationPoints[3].y);
00278     ySlope = (slope1 + slope2) / 2;
00279 
00280     /* Test point */
00281     LCDD_Fill(0xFFFFFF);
00282     LCDD_DrawString(30, 50, (uint8_t *)"LCD calibration", COLOR_BLACK);
00283     LCDD_DrawString(1, 100, (uint8_t *)" Touch the point to\nvalidate calibration", COLOR_DARKBLUE);
00284     DrawCalibrationPoint(&testPoint);
00285 
00286     /* Wait for touch & end of conversion */
00287     TSD_WaitPenPressed();
00288 
00289     TSD_GetRawMeasurement(measuredPoint.data);
00290     TSDCom_InterpolateMeasurement(measuredPoint.data, (uint32_t *) &measuredPoint);
00291     DrawCalibrationPoint(&measuredPoint);
00292 
00293     /* Check resulting x and y */
00294     xDiff = (int32_t) measuredPoint.x - (int32_t) testPoint.x;
00295     yDiff = (int32_t) measuredPoint.y - (int32_t) testPoint.y;
00296     xOk = (xDiff >= -POINTS_MAX_ERROR) && (xDiff <= POINTS_MAX_ERROR);
00297     yOk = (yDiff >= -POINTS_MAX_ERROR) && (yDiff <= POINTS_MAX_ERROR);
00298 
00299     /* Wait for contact loss */
00300     TSD_WaitPenReleased();
00301 
00302     /* Check calibration result */
00303     if (xOk && yOk) {
00304 
00305         bCalibrationOk = 1;
00306         LCDD_Fill(COLOR_WHITE);
00307         LCDD_DrawString(30, 50, (uint8_t *)"LCD calibration", COLOR_BLACK);
00308         LCDD_DrawString(80, 140, (uint8_t *)"Success !", COLOR_GREEN);
00309 
00310     }
00311     else {
00312 
00313         bCalibrationOk = 0;
00314         LCDD_Fill(COLOR_WHITE);
00315         LCDD_DrawString(30, 50, (uint8_t *)"LCD calibration", COLOR_BLACK);
00316         LCDD_DrawString(40, 140, (uint8_t *)"Error too big", COLOR_RED);
00317     }
00318 
00319     /* Slight delay */
00320     for (i = 0; i < DELAY_RESULT_DISPLAY; i++);
00321 
00322     return (xOk && yOk);
00323 }
