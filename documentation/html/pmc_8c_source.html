00049 /**
00050  * \brief Enables the clock of a peripheral. The peripheral ID is used
00051  * to identify which peripheral is targetted.
00052  *
00053  * \note The ID must NOT be shifted (i.e. 1 << ID_xxx).
00054  *
00055  * \param id  Peripheral ID (ID_xxx).
00056  */
00057 extern void PMC_EnablePeripheral( uint32_t dwId )
00058 {
00059     assert( dwId < 35 ) ;
00060 
00061     if ( dwId < 32 )
00062     {
00063         if ( (PMC->PMC_PCSR0 & ((uint32_t)1 << dwId)) == ((uint32_t)1 << dwId) )
00064         {
00065             TRACE_DEBUG( "PMC_EnablePeripheral: clock of peripheral"  " %u is already enabled\n\r", dwId ) ;
00066         }
00067         else
00068         {
00069             PMC->PMC_PCER0 = 1 << dwId ;
00070         }
00071     }
00072     else
00073     {
00074         dwId -= 32;
00075         if ((PMC->PMC_PCSR1 & ((uint32_t)1 << dwId)) == ((uint32_t)1 << dwId))
00076         {
00077             TRACE_DEBUG( "PMC_EnablePeripheral: clock of peripheral"  " %u is already enabled\n\r", dwId + 32 ) ;
00078         }
00079         else
00080         {
00081             PMC->PMC_PCER1 = 1 << dwId ;
00082         }
00083     }
