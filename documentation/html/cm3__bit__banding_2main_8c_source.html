00032 /**
00033  * \page cm3_bit_banding Bit Banding for SAM CM3 Microcontrollers
00034  *
00035  * \section Purpose
00036  *
00037  * The cm3-bitbanding example will help new users get familiar with
00038  * bitbanding feature of SAM3 microcontrollers.
00039  *
00040  * Bit-banding a useful feature of Cortex M3 processor. The processor memory map
00041  * includes two bit-band regions, that occupy the lowest 1MB of the SRAM and
00042  * Peripheral memory regions. These bit-band regions map each word in an alias
00043  * region of memory to a bit in a bit-band region of memory. Accessing to the
00044  * alias region map to the corresponding 1MB memory region map word applies to
00045  * the bit that mapped.
00046  *
00047  * With bit-banding, bits modification can be simplified to just memory write,
00048  * so that the bit modification is easier and faster
00049  * (see result of BitbandingSramTest()):
00050  * - Old bit modification (see WRITE_BIT())
00051  *    -# Read word from memory
00052  *    -# Mask out the bit that is to be modified
00053  *    -# Set the expected bit to right value
00054  *    -# Write word back to memory
00055  * - Bit modification using bit-banding (see WRITE_BITBANDING())
00056  *    -# Calculate bit alias word address
00057  *    -# Write to address memory with expected value
00058  *
00059  * The example will execute following tests:
00060  * - Test bit accessing speed with a sram buffer
00061  *   (BitbandingSramTest() & testData[])
00062  * - Test REGISTER bit-banding accessing with LED0
00063  *   (BitbandingPeripheralTest())
00064  *
00065  * \section Requirements
00066  *
00067  * This package can be used with SAM3S evaluation kits.
00068  *
00069  * \section Description
00070  * The cm3-bitbanding example will help new users get familiar with
00071  * bitbanding feature of SAM3 microcontrollers.
00072  *
00073  * \section Usage
00074  *
00075  * -# Build the program and download it inside the evaluation board. Please
00076  *    refer to the
00077  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6224.pdf">
00078  *    SAM-BA User Guide</a>, the
00079  *    <a href="http://www.atmel.com/dyn/resources/prod_documents/doc6310.pdf">
00080  *    GNU-Based Software Development</a> application note or to the
00081  *    <a href="ftp://ftp.iar.se/WWWfiles/arm/Guides/EWARM_UserGuide.ENU.pdf">
00082  *    IAR EWARM User Guide</a>, depending on your chosen solution.
00083  * -# On the computer, open and configure a terminal application
00084  *    (e.g. HyperTerminal on Microsoft Windows) with these settings:
00085  *   - 115200 bauds
00086  *   - 8 bits of data
00087  *   - No parity
00088  *   - 1 stop bit
00089  *   - No flow control
00090  * -# Start the application.
00091  * -# In the terminal window, the following text should appear
00092  *     (values depend on the board and chip used):
00093  *    \code
00094  *     -- CM3 Bit Banding Example xxx --
00095  *     -- xxxxxx-xx
00096  *     -- Compiled: xxx xx xxxx xx:xx:xx --

00128 /**
00129  * \brief Normal bit write macro
00130  *
00131  * Write bit through normal bit accessing mode.
00132  * \param addr32 32-bit aligned byte address where the bit exists.
00133  * \param bit    Bit position in the DW.
00134  * \param val    The value that the bit is set to.
00135  */
00136 #define WRITE_BIT(addr32, bit, val) do {\
00137         register uint32_t tmp = *(uint32_t*)(addr32); \
00138         tmp &= ~(1     << (bit)); \
00139         tmp |=  ((val) << (bit)); \
00140         *(uint32_t*)(addr32) = tmp; \
00141     } while (0);

00163  * \brief Perform bitbanding tests in SRAM
00164  *
00165  * Compare writing to SRAM bits with bitbanding against normal bit access.
00166  * -# Record start tick via GetTickCount().
00167  * -# Modify bits in testData[] bit by bit for NB_TEST_ROUND times.
00168  * -# Record end tick via GetTickCount().

00170  *
00171  * \callgraph
00172  */
00173 static void _BitbandingSramTest( void )
00174 {
00175     uint32_t i, j, nbTests ;
00176     uint32_t startTick, endTick ;
00177 
00178     printf( "\n\r** Modify SRAM with bitbanding:\n\r" ) ;
00179     if ( !IS_BITBAND_SRAM_ADDR( testData ) )
00180     {
00181         TRACE_FATAL("Data Address %x not in region!!", (int)testData);
00182     }
00183 
00184     /* Bit accessing speed compare */
00185     printf( "-- BITs Writing performance test:\n\r" ) ;
00186 
00187     /* - Write to bits with normal bit accessing */
00188     printf( " Without bitbanding:" ) ;
00189     Wait(1); /* Start on tick point! */
00190     startTick = GetTickCount();
00191     for (nbTests = 0; nbTests < NB_TEST_ROUND; nbTests ++)
00192     {
00193         for (i = 0; i < TEST_DATA_SIZE; i ++)
00194         {
00195             for (j = 0; j < 32; j ++)
00196             {
00197                 WRITE_BIT(&testData[i] , j, j & 0x1);
00198             }
00199         }
00200     }

00203
